package apoMario.game.panels;

import java.io.*;
import java.nio.file.Path;
import java.util.*;
import apoMario.level.ApoMarioLevel;

public class ApoMarioStateAchievements {
    private static final String POINTS_GOAL = "POINTS_GOAL";
    private static final String ENEMIES_GOAL = "ENEMIES_GOAL";
    private static final String TIME_GOAL = "TIME_GOAL";

    private final Path store;
    private Set<String> unlockedAchievements;
    
    private int currentRunPoints;
    private int currentRunEnemies;
    private long currentRunTime;

    public ApoMarioStateAchievements(Path store) {
        this.store = store;
        this.unlockedAchievements = new HashSet<>();
        this.load();
    }

    public void beginRun() {
        this.currentRunPoints = 0;
        this.currentRunEnemies = 0;
        this.currentRunTime = 0;
    }

    public void addPoints(int points) {
        this.currentRunPoints += points;
        if (this.currentRunPoints >= 20000) {
            this.unlockedAchievements.add(POINTS_GOAL);
        }
    }

    public void addEnemyKilled() {
        this.currentRunEnemies++;
        if (this.currentRunEnemies >= 3) {
            this.unlockedAchievements.add(ENEMIES_GOAL);
        }
    }

    public void addTimeSurvived(long ms) {
        this.currentRunTime += ms;
        if (this.currentRunTime >= 60000) {
            this.unlockedAchievements.add(TIME_GOAL);
        }
    }

    public void recordRunEnd(ApoMarioLevel level) {
        // Final evaluation against level data
        if (this.currentRunPoints >= 20000) this.unlockedAchievements.add(POINTS_GOAL);
        if (this.currentRunEnemies >= 3) this.unlockedAchievements.add(ENEMIES_GOAL);
        if (this.currentRunTime >= 60000) this.unlockedAchievements.add(TIME_GOAL);
        this.saveAcrossSessions();
    }

    public List<String> getUnlockedAchievements() {
        return new ArrayList<>(this.unlockedAchievements);
    }

    public void saveAcrossSessions() {
        try (BufferedWriter writer = new BufferedWriter(new FileWriter(store.toFile()))) {
            for (String ach : unlockedAchievements) {
                writer.write(ach);
                writer.newLine();
            }
        } catch (IOException e) {
            e.printStackTrace();
        }
    }

    private void load() {
        File file = store.toFile();
        if (!file.exists()) return;
        try (BufferedReader reader = new BufferedReader(new FileReader(file))) {
            String line;
            while ((line = reader.readLine()) != null) {
                unlockedAchievements.add(line);
            }
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}

    public void die(ApoMarioLevel level, int player) {
        if (player >= 0) {
            level.getPlayers().get(player).setPoints(level.getPlayers().get(player).getPoints() + this.points);
        }
        // Track kill
        if (level.getAchievements() != null) {
            level.getAchievements().addEnemyKilled();
        }
        this.setBDie(true);
        this.setVecY(-ApoMarioConstants.ENEMY_VEC_Y_DIE);
    }
    // ... (rest of class)
}
