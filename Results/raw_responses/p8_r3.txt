To integrate the Highscore feature, we will create a data-handling class `ApoMarioHighscore` and a UI panel `ApoMarioHighscorePanel`. These will be integrated into the existing `ApoMarioLevel` (for tracking) and `ApoMarioPanel` (for storage and navigation) and `ApoMarioMenu` (for access).

```java filename=apoMario/game/panels/ApoMarioHighscore.java
package apoMario.game.panels;

import java.io.*;
import java.nio.file.Path;
import java.util.*;
import apoMario.level.ApoMarioLevel;
import apoMario.entity.ApoMarioPlayer;

public class ApoMarioHighscore {
    private Path store;
    private List<String> names = new ArrayList<>();
    private List<Integer> scores = new ArrayList<>();
    private List<Integer> survivalTimes = new ArrayList<>();

    public ApoMarioHighscore(Path store) {
        this.store = store;
        load();
    }

    public boolean storeRun(int score, int survivalTime, String playerName) {
        names.add(playerName);
        scores.add(score);
        survivalTimes.add(survivalTime);
        sort();
        persistAcrossRuns();
        return true;
    }

    private void sort() {
        List<Integer> indices = new ArrayList<>();
        for (int i = 0; i < scores.size(); i++) indices.add(i);
        indices.sort((a, b) -> scores.get(b).compareTo(scores.get(a)));

        List<String> sortedNames = new ArrayList<>();
        List<Integer> sortedScores = new ArrayList<>();
        List<Integer> sortedTimes = new ArrayList<>();

        for (int i : indices) {
            sortedNames.add(names.get(i));
            sortedScores.add(scores.get(i));
            sortedTimes.add(survivalTimes.get(i));
        }
        names = sortedNames;
        scores = sortedScores;
        survivalTimes = sortedTimes;
    }

    public void persistAcrossRuns() {
        try (PrintWriter writer = new PrintWriter(new FileWriter(store.toFile()))) {
            for (int i = 0; i < names.size(); i++) {
                writer.println(names.get(i) + "," + scores.get(i) + "," + survivalTimes.get(i));
            }
        } catch (IOException e) {
            e.printStackTrace();
        }
    }

    private void load() {
        if (!store.toFile().exists()) return;
        try (BufferedReader reader = new BufferedReader(new FileReader(store.toFile()))) {
            String line;
            while ((line = reader.readLine()) != null) {
                String[] parts = line.split(",");
                if (parts.length == 3) {
                    names.add(parts[0]);
                    scores.add(Integer.parseInt(parts[1]));
                    survivalTimes.add(Integer.parseInt(parts[2]));
                }
            }
        } catch (Exception e) {
            e.printStackTrace();
        }
    }

    public void recordRunEnd(ApoMarioLevel level) {
        int timeInSeconds = level.getPassedTime() / 1000;
        for (ApoMarioPlayer player : level.getPlayers()) {
            String name = "Human";
            if (player.getAi() != null && player.getAi().getTeamName() != null) {
                name = player.getAi().getTeamName();
            }
            storeRun(player.getPoints(), timeInSeconds, name);
        }
    }

    public List<String> getPlayersNames() { return names; }
    public List<Integer> getPlayersScores() { return scores; }
    public List<Integer> getSurvivalTimes() { return survivalTimes; }
}
```

```java filename=apoMario/game/panels/ApoMarioHighscorePanel.java
package apoMario.game.panels;

import java.awt.Color;
import java.awt.Graphics2D;
import java.awt.event.KeyEvent;
import java.util.List;
import apoMario.ApoMarioConstants;
import apoMario.game.ApoMarioPanel;

public class ApoMarioHighscorePanel extends ApoMarioModel {
    private ApoMarioPanel game;

    public ApoMarioHighscorePanel(ApoMarioPanel game) {
        this.game = game;
    }

    @Override
    public void init() {}

    @Override
    public void think(int delta) {}

    @Override
    public void render(Graphics2D g) {
        g.setColor(Color.WHITE);
        g.fillRect(0, 0, ApoMarioConstants.GAME_WIDTH, ApoMarioConstants.GAME_HEIGHT);
        g.setColor(Color.BLACK);
        g.setFont(ApoMarioConstants.FONT_MENU);
        
        g.drawString("Highscores", 50, 50);
        
        List<String> names = game.getHighscore().getPlayersNames();
        List<Integer> scores = game.getHighscore().getPlayersScores();
        List<Integer> times = game.getHighscore().getSurvivalTimes();
        
        int y = 100;
        for (int i = 0; i < Math.min(10, names.size()); i++) {
            int m = times.get(i) / 60;
            int s = times.get(i) % 60;
            String timeStr = String.format("%02d:%02d", m, s);
            g.drawString((i + 1) + ". " + names.get(i) + " - " + scores.get(i) + " pts - " + timeStr, 50, y);
            y += 30;
        }
        g.drawString("Press ESC to return", 50, ApoMarioConstants.GAME_HEIGHT - 50);
    }

    @Override
    public void keyButtonReleased(int button, char character) {
        if (button == KeyEvent.VK_ESCAPE) {
            game.setMenu();
        }
    }

    @Override
    public void mouseButtonReleased(int x, int y) {}

    @Override
    public void mousePressed(int x, int y, boolean bRight) {}

    @Override
    public void mouseMoved(int x, int y) {}

    @Override
    public boolean mouseDragged(int x, int y) { return false; }
}
```
