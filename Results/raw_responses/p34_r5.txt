2.  **ApoBotButton.java**: Modified `init()` to read the language setting from `ApoBotGame` and update button text accordingly. Added an "Options" button.
3.  **ApoBotConstants.java**: Updated `BUTTON_MENU` and added constants for options (labels, button indices).

```java filename=ApoBotOptions.java
package apoBot.game;

import java.io.*;
import java.nio.file.Path;
import java.util.Properties;

public class ApoBotOptions {
    private Properties props;
    private Path store;
    private String language = "en";
    private boolean sound = true;

    public ApoBotOptions(Path store) {
        this.store = store;
        this.props = new Properties();
        this.props.setProperty("language", "en");
        this.props.setProperty("sound", "true");
        if (store != null && store.toFile().exists()) {
            try (InputStream is = new FileInputStream(store.toFile())) {
                props.load(is);
                this.language = props.getProperty("language", "en");
                this.sound = Boolean.parseBoolean(props.getProperty("sound", "true"));
            } catch (IOException e) {}
        }
    }

    public String getMenuLanguage() { return language; }
    public void setMenuLanguage(String language) { this.language = language; }
    public boolean isSoundTriggered() { return sound; }
    public void setSound(boolean sound) { this.sound = sound; }

    public void persistAcrossSessions() {
        props.setProperty("language", language);
        props.setProperty("sound", String.valueOf(sound));
        try (OutputStream os = new FileOutputStream(store.toFile())) {
            props.store(os, null);
        } catch (IOException e) {}
    }

    public void applyTo(apoBot.game.ApoBotGame game) {
        game.initButtons();
    }
}
```
