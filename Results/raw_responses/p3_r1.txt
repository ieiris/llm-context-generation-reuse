The Highscore feature has been implemented by creating a new `ApoMarioHighscore` data management class, an `ApoMarioHighscorePanel` for the UI, and integrating these into the game's existing structure.

1.  **`ApoMarioHighscore.java`**: Implements the logic for storing, sorting, and persisting highscore data.
2.  **`ApoMarioHighscorePanel.java`**: Implements the UI panel for displaying the highscore board, reusing existing rendering and layout patterns from the menu.
3.  **`ApoMarioLevel.java`**: Modified to call `recordRunEnd` in the `setAnalysis` method, which is the natural end-of-run trigger.
4.  **`ApoMarioPanel.java`**: Modified to initialize the highscore system and manage navigation to the new highscore panel.
5.  **`ApoMarioMenu.java`**: Modified to handle the new "Highscore" button click event.
6.  **`ApoMarioButtons.java`**: Modified to include a new button for the Highscore feature in the main menu.

```java filename=ApoMarioHighscore.java
package apoMario.game.panels;

import java.io.BufferedReader;
import java.io.BufferedWriter;
import java.io.IOException;
import java.nio.file.Files;
import java.nio.file.Path;
import java.util.ArrayList;
import java.util.Collections;
import java.util.Comparator;
import java.util.List;

import apoMario.level.ApoMarioLevel;

public class ApoMarioHighscore {

    private final Path store;
    private final List<String> names = new ArrayList<>();
    private final List<Integer> scores = new ArrayList<>();
    private final List<Integer> times = new ArrayList<>();

    public ApoMarioHighscore(Path store) {
        this.store = store;
        load();
    }

    public boolean storeRun(int score, int survivalTime, String playerName) {
        names.add(playerName);
        scores.add(score);
        times.add(survivalTime);
        sort();
        return true;
    }

    public List<String> getPlayersNames() {
        return names;
    }

    public List<Integer> getPlayersScores() {
        return scores;
    }

    public List<Integer> getSurvivalTimes() {
        return times;
    }

    public void persistAcrossRuns() {
        try (BufferedWriter writer = Files.newBufferedWriter(store)) {
            for (int i = 0; i < names.size(); i++) {
                writer.write(names.get(i) + "," + scores.get(i) + "," + times.get(i));
                writer.newLine();
            }
        } catch (IOException e) {
            e.printStackTrace();
        }
    }

    public void recordRunEnd(ApoMarioLevel level) {
        int score = 0;
        String playerName = "Player";
        if (level.getPlayers() != null && !level.getPlayers().isEmpty()) {
            score = level.getPlayers().get(0).getPoints();
            if (level.getPlayers().get(0).getTeamName() != null) {
                playerName = level.getPlayers().get(0).getTeamName();
            }
        }
        int time = level.getPassedTime(); // already in ms
        storeRun(score, time, playerName);
        persistAcrossRuns();
    }

    private void load() {
        if (!Files.exists(store)) return;
        try (BufferedReader reader = Files.newBufferedReader(store)) {
            String line;
            while ((line = reader.readLine()) != null) {
                String[] parts = line.split(",");
                if (parts.length == 3) {
                    names.add(parts[0]);
                    scores.add(Integer.parseInt(parts[1]));
                    times.add(Integer.parseInt(parts[2]));
                }
            }
            sort();
        } catch (IOException | NumberFormatException e) {
            e.printStackTrace();
        }
    }

    private void sort() {
        List<Integer> indices = new ArrayList<>();
        for (int i = 0; i < scores.size(); i++) indices.add(i);
        
        Collections.sort(indices, new Comparator<Integer>() {
            @Override
            public int compare(Integer i1, Integer i2) {
                return scores.get(i2).compareTo(scores.get(i1));
            }
        });

        List<String> sortedNames = new ArrayList<>();
        List<Integer> sortedScores = new ArrayList<>();
        List<Integer> sortedTimes = new ArrayList<>();
        
        for (int index : indices) {
            sortedNames.add(names.get(index));
            sortedScores.add(scores.get(index));
            sortedTimes.add(times.get(index));
        }
        
        names.clear(); names.addAll(sortedNames);
        scores.clear(); scores.addAll(sortedScores);
        times.clear(); times.addAll(sortedTimes);
    }
}
```

```java filename=ApoMarioHighscorePanel.java
package apoMario.game.panels;

import java.awt.Color;
import java.awt.Graphics2D;
import java.util.List;

import apoMario.ApoMarioConstants;
import apoMario.game.ApoMarioPanel;

public class ApoMarioHighscorePanel extends ApoMarioModelMenu {

    public static final String FUNCTION_BACK = "highscore_back";

    public ApoMarioHighscorePanel(ApoMarioPanel game) {
        super(game);
    }

    @Override
    public void init() {
        super.init();
    }

    @Override
    public void render(Graphics2D g) {
        super.renderBackgroundAndAnimation(g);
        
        g.setFont(ApoMarioConstants.FONT_MENU);
        g.setColor(Color.BLACK);
        
        String title = "Highscore";
        int w = g.getFontMetrics().stringWidth(title);
        g.drawString(title, ApoMarioConstants.GAME_WIDTH / 2 - w / 2, 100);
        
        ApoMarioHighscore highscore = this.getGame().getHighscore();
        List<String> names = highscore.getPlayersNames();
        List<Integer> scores = highscore.getPlayersScores();
        List<Integer> times = highscore.getSurvivalTimes();
        
        int y = 150;
        for (int i = 0; i < names.size() && i < 10; i++) {
            String entry = (i + 1) + ". " + names.get(i) + " - " + scores.get(i) + " pts - " + (times.get(i) / 60000) + ":" + ((times.get(i) / 1000) % 60);
            g.drawString(entry, 100, y);
            y += 30;
        }
        
        super.renderButtonsAndRunner(g);
    }

    @Override
    public void mouseButtonFunction(String function) {
        if (function.equals(FUNCTION_BACK)) {
            this.getGame().setMenu();
        }
    }

    @Override
    public void keyButtonReleased(int button, char character) {
        if (button == java.awt.event.KeyEvent.VK_ESCAPE) {
            this.getGame().setMenu();
        }
    }

    @Override
    public void excecuteFunction() {}
}
```
