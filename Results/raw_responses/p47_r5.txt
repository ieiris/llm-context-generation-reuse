```java filename=ApoBotOptions.java
package apoBot.game;

import java.io.*;
import java.nio.file.Path;
import java.util.Properties;

public class ApoBotOptions {
    private Properties properties = new Properties();
    private Path store;

    public ApoBotOptions(Path store) {
        this.store = store;
        load();
    }

    private void load() {
        try (InputStream is = new FileInputStream(store.toFile())) {
            properties.load(is);
        } catch (IOException e) {
            properties.setProperty("language", "en");
            properties.setProperty("sound", "true");
        }
    }

    public void persistAcrossSessions() {
        try (OutputStream os = new FileOutputStream(store.toFile())) {
            properties.store(os, "ApoBot Options");
        } catch (IOException e) {
        }
    }

    public String getMenuLanguage() {
        return properties.getProperty("language", "en");
    }

    public void setMenuLanguage(String language) {
        properties.setProperty("language", language);
        persistAcrossSessions();
    }

    public boolean isSoundTriggered() {
        return Boolean.parseBoolean(properties.getProperty("sound", "true"));
    }

    public void setSound(boolean sound) {
        properties.setProperty("sound", String.valueOf(sound));
        persistAcrossSessions();
    }

    public void applyTo(ApoBotGame game) {
        // Re-initialize buttons to apply language
        game.setButtons(null);
        new ApoBotButton(game).init();
    }
}
```
