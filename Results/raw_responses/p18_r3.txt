The Achievements feature has been integrated into the ApoMario game.

### Integration Steps:
1.  **`ApoMarioStateAchievements.java`**: Created to track game milestones (POINTS_GOAL, ENEMIES_GOAL, TIME_GOAL) and manage persistence.
2.  **`ApoMarioComponent.java`**: Modified to include `ApoMarioStateAchievements` and the `getAchievements()` method.
3.  **`ApoMarioPanel.java`**: Modified to initialize the achievements system and handle switching to the achievements view.
4.  **`ApoMarioLevel.java`**: Modified to track run statistics and trigger achievement evaluations at the end of a run.
5.  **`ApoMarioPlayer.java`**: Modified to pass point updates and enemy kill events to the achievements system.
6.  **`ApoMarioMenu.java`**: Modified to handle navigation to the achievements screen.

public class ApoMarioStateAchievements extends ApoMarioModelMenu {

    public static final String POINTS_GOAL = "POINTS_GOAL";
    public static final String ENEMIES_GOAL = "ENEMIES_GOAL";
    public static final String TIME_GOAL = "TIME_GOAL";

    private List<String> unlocked;
    private Path store;
    
    private int currentPoints;
    private int currentEnemies;
    private long currentTime;

    public ApoMarioStateAchievements(ApoMarioPanel game, Path store) {
        super(game);
        this.store = store;
        this.unlocked = new ArrayList<String>();
        this.loadAchievements();
    }

    public void beginRun() {
        this.currentPoints = 0;
        this.currentEnemies = 0;
        this.currentTime = 0;
    }

    public void addPoints(int points) {
        this.currentPoints += points;
        this.checkGoals();
    }

    public void addEnemyKilled() {
        this.currentEnemies++;
        this.checkGoals();
    }

    public void addTimeSurvived(long ms) {
        this.currentTime = ms;
        this.checkGoals();
    }

    public void recordRunEnd(ApoMarioLevel level) {
        this.checkGoals();
        this.saveAcrossSessions();
    }

    private void checkGoals() {
        if (this.currentPoints >= 20000 && !this.unlocked.contains(POINTS_GOAL)) {
            this.unlocked.add(POINTS_GOAL);
        }
        if (this.currentEnemies >= 3 && !this.unlocked.contains(ENEMIES_GOAL)) {
            this.unlocked.add(ENEMIES_GOAL);
        }
        if (this.currentTime >= 60000 && !this.unlocked.contains(TIME_GOAL)) {
            this.unlocked.add(TIME_GOAL);
        }
    }

    public List<String> getUnlockedAchievements() {
        return new ArrayList<String>(this.unlocked);
    }

    public void saveAcrossSessions() {
        try (ObjectOutputStream oos = new ObjectOutputStream(new FileOutputStream(store.toFile()))) {
            oos.writeObject(this.unlocked);
        } catch (IOException e) {
            e.printStackTrace();
        }
    }

    @SuppressWarnings("unchecked")
    private void loadAchievements() {
        if (store.toFile().exists()) {
            try (ObjectInputStream ois = new ObjectInputStream(new FileInputStream(store.toFile()))) {
                this.unlocked = (List<String>) ois.readObject();
            } catch (Exception e) {
                e.printStackTrace();
            }
        }
    }

    @Override
    public void init() {}

    @Override
    public void makeBackground() {}

    @Override
    public void makeRunner() {}

    @Override
    public void makeBackgroundAnimation() {}

    @Override
    public void makeSearch() {}

    @Override
    public void excecuteFunction() {
        if (getExcecuteFunction().equals("back")) {
            getGame().setMenu();
        }
    }

    @Override
    public void keyButtonReleased(int button, char character) {
        if (button == KeyEvent.VK_ESCAPE) {
            getGame().setMenu();
        }
    }

    @Override
    public void mouseButtonFunction(String function) {
        setExcecuteFunction(function);
        excecuteFunction();
    }

    @Override
    public void mouseButtonReleased(int x, int y) {}

    @Override
    public boolean mouseMoved(int x, int y) { return false; }

    @Override
    public boolean mouseDragged(int x, int y) { return false; }

    @Override
    public boolean mousePressed(int x, int y, boolean bRight) { return false; }

    @Override
    public void think(int delta) {}

    @Override
    public void render(Graphics2D g) {
        g.setColor(Color.WHITE);
        g.fillRect(0, 0, ApoMarioConstants.GAME_WIDTH, ApoMarioConstants.GAME_HEIGHT);
        g.setColor(Color.BLACK);
        g.setFont(ApoMarioConstants.FONT_MENU);
        g.drawString("Achievements", 50, 50);
        
        int y = 100;
        for (String ach : new String[]{POINTS_GOAL, ENEMIES_GOAL, TIME_GOAL}) {
            String status = unlocked.contains(ach) ? "[UNLOCKED]" : "[LOCKED]";
            g.drawString(ach + ": " + status, 50, y);
            y += 50;
        }
        g.drawString("Press ESC to return", 50, y + 50);
    }
}

    public synchronized void think(int delta) {
        if ((this.bInit) || (this.isBSimulate())) { return; }
        this.middleCount.add(this.renderTime);
        if (this.middleCount.size() >= 100) {
            this.middleCount.remove(0);
            long middle = 0;
            for (int i = 0; i < this.middleCount.size(); i++) { middle += this.middleCount.get(i); }
            middle = middle / (long)this.middleCount.size();
            if (middle > 100000000l) { this.setBMiddle(false); }
            this.middleCount.clear();
        }
        if (this.isBPause()) { this.curTime += delta; if (this.curTime > ApoMarioReplayPlay.CHANGE_TIME) { this.curTime -= ApoMarioReplayPlay.CHANGE_TIME; this.bShow = !this.bShow; } }
        long t = System.nanoTime();
        super.think(delta);
        this.thinkLevelGame(delta);
        this.thinkTime = (int) (System.nanoTime() - t);
    }

    public synchronized void thinkLevelGame(int delta) { if (this.model != null) { this.model.think(delta); } }

    public synchronized void render(Graphics2D g) {
        long t = System.nanoTime();
        Shape s = g.getClip();
        g.setClip(0, 0, ApoMarioConstants.GAME_WIDTH, ApoMarioConstants.GAME_HEIGHT);
        try {
            if (this.model != null) { this.model.render(g); }
            if (!this.bSimulate) {
                if (this.model != null) { this.model.render(g); }
                if ((this.model != null) && ((this.model instanceof ApoMarioGame))) {
                    if (this.bMiniMap) { this.miniMap.render(g); } else { this.renderLevelWidthStatistics(g); }
                    this.renderPause(g);
                }
                this.renderFPS(g);
            }
            this.renderSimulate(g);
        } catch (Exception ex) { ex.printStackTrace(); }
        g.setClip(s);
        this.renderTime = (int)(System.nanoTime() - t);
    }

    private void renderSimulate(Graphics2D g) {
        if (this.bSimulate) {
            g.setColor(Color.WHITE);
            int width = 200 * ApoMarioConstants.SIZE * ApoMarioConstants.APP_SIZE;
            int height = 50 * ApoMarioConstants.SIZE * ApoMarioConstants.APP_SIZE;

            g.drawString(s, ApoMarioConstants.GAME_WIDTH/2 - w/2, y + height/2);
            s = "Round: "+this.simulation.getCurRound()+" von "+this.simulation.getCount();
            w = g.getFontMetrics().stringWidth(s);
            g.drawString(s, ApoMarioConstants.GAME_WIDTH/2 - w/2, y + height/2 + g.getFontMetrics().getHeight() + 1 * 5);
        }
    }

    private void renderLevelWidthStatistics(Graphics2D g) {
        g.setFont(ApoMarioConstants.FONT_STATISTICS);
        int difWidth = 20;
        int levelWidth = ApoMarioConstants.GAME_WIDTH - difWidth * 2;
        float maxWidth = -1;
        for (int i = 0; i < this.getLevel().getPlayers().size(); i++) {
            ApoMarioPlayer player = this.getLevel().getPlayers().get(i);
            if (player.isBVisible()) {
                float marioWidth = (player.getX() * 100f / (this.getLevel().getWidth() * ApoMarioConstants.TILE_SIZE));
                if (maxWidth < marioWidth) { maxWidth = marioWidth; }
                if (i == 0) { g.setColor(Color.RED); } else { g.setColor(Color.BLUE); }
                g.drawString("M", difWidth + levelWidth * marioWidth / 100, ApoMarioConstants.GAME_HEIGHT - 5);
            }
        }
        int distance = 15 * ApoMarioConstants.SIZE;
        for (int i = 0; i < levelWidth / distance; i++) {
            int width = i * distance;
            if ((float)width / (float)levelWidth < maxWidth / 100f) { g.setColor(Color.GREEN); } else { g.setColor(Color.BLACK); }
            g.fillRect(difWidth + width, ApoMarioConstants.GAME_HEIGHT - 10, 5, 5);
        }
        String s = "F";
        int w = g.getFontMetrics().stringWidth(s);
        g.drawString(s, difWidth + levelWidth - w, ApoMarioConstants.GAME_HEIGHT - 5);
    }

    private void renderFPS(Graphics2D g) {
        if (ApoMarioConstants.FPS) {
            g.setColor(Color.WHITE);
            g.setFont(ApoMarioConstants.FONT_FPS);
            g.drawString("think time: " + this.thinkTime + " ns", 5 + 1, ApoMarioConstants.GAME_HEIGHT - 45 + 1);
            g.drawString("think time: " + (this.thinkTime / 1000000) + " ms",5 + 1, ApoMarioConstants.GAME_HEIGHT - 35 + 1);

            g.drawString("draw time: " + (this.renderTime / 1000000) + " ms", 5 + 1, ApoMarioConstants.GAME_HEIGHT - 15 + 1);
            g.drawString("FPS: " + super.getFPS(), 5 + 1, ApoMarioConstants.GAME_HEIGHT - 5 + 1);
            g.setColor(Color.BLACK);
            g.drawString("think time: " + this.thinkTime + " ns", 5, ApoMarioConstants.GAME_HEIGHT - 45);
            g.drawString("think time: " + (this.thinkTime / 1000000) + " ms",5, ApoMarioConstants.GAME_HEIGHT - 35);
            g.drawString("draw time: " + this.renderTime + " ns", 5, ApoMarioConstants.GAME_HEIGHT - 25);

            g.drawString("FPS: " + super.getFPS(), 5, ApoMarioConstants.GAME_HEIGHT - 5);
        }
    }

            g.setColor(Color.BLACK);
            g.drawString(s, ApoMarioConstants.GAME_WIDTH/2 - w/2, hei + 20 * ApoMarioConstants.SIZE);
        }
    }
