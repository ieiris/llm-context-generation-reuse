package apoMario.game.panels;

import java.io.BufferedReader;
import java.io.BufferedWriter;
import java.io.IOException;
import java.nio.file.Files;
import java.nio.file.Path;
import java.util.ArrayList;
import java.util.List;

import apoMario.level.ApoMarioLevel;

public class ApoMarioStateAchievements {

    public static final String POINTS_GOAL = "POINTS_GOAL";
    public static final String ENEMIES_GOAL = "ENEMIES_GOAL";
    public static final String TIME_GOAL = "TIME_GOAL";

    private final Path store;
    private List<String> unlockedAchievements;
    private int currentPoints;
    private int currentEnemies;
    private long currentTime;

    public ApoMarioStateAchievements(java.nio.file.Path store) {
        this.store = store;
        this.unlockedAchievements = new ArrayList<>();
        this.loadAcrossSessions();
    }

    public void beginRun() {
        this.currentPoints = 0;
        this.currentEnemies = 0;
        this.currentTime = 0;
    }

    public void addPoints(int points) {
        this.currentPoints += points;
    }

    public void addEnemyKilled() {
        this.currentEnemies++;
    }

    public void addTimeSurvived(long ms) {
        this.currentTime += ms;
    }

    public void recordRunEnd(ApoMarioLevel level) {
        if (currentPoints >= 20000 && !unlockedAchievements.contains(POINTS_GOAL)) {
            unlockedAchievements.add(POINTS_GOAL);
        }
        if (currentEnemies >= 3 && !unlockedAchievements.contains(ENEMIES_GOAL)) {
            unlockedAchievements.add(ENEMIES_GOAL);
        }
        if (currentTime >= 60000 && !unlockedAchievements.contains(TIME_GOAL)) {
            unlockedAchievements.add(TIME_GOAL);
        }
        saveAcrossSessions();
    }

    public List<String> getUnlockedAchievements() {
        return new ArrayList<>(unlockedAchievements);
    }

    public void saveAcrossSessions() {
        try (BufferedWriter writer = Files.newBufferedWriter(store)) {
            for (String achievement : unlockedAchievements) {
                writer.write(achievement);
                writer.newLine();
            }
        } catch (IOException e) {
            e.printStackTrace();
        }
    }

    private void loadAcrossSessions() {
        if (Files.exists(store)) {
            try (BufferedReader reader = Files.newBufferedReader(store)) {
                String line;
                while ((line = reader.readLine()) != null) {
                    unlockedAchievements.add(line);
                }
            } catch (IOException e) {
                e.printStackTrace();
            }
        }
    }
}
