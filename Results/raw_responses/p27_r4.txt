To integrate the Achievements feature into ApoMario, we will create the `ApoMarioStateAchievements` class to manage the state and persistence of achievements, and the `ApoMarioAchievementsPanel` class to display them in the game menu. We will then wire this into the existing game flow by modifying `ApoMarioLevel`, `ApoMarioEnemy`, `ApoMarioMenu`, and `ApoMarioPanel`.

```java filename=apoMario/game/panels/ApoMarioStateAchievements.java
package apoMario.game.panels;

import java.io.FileInputStream;
import java.io.FileOutputStream;
import java.io.IOException;
import java.nio.file.Path;
import java.util.ArrayList;
import java.util.List;
import java.util.Properties;

import apoMario.level.ApoMarioLevel;

public class ApoMarioStateAchievements {
    private static final String POINTS_GOAL = "POINTS_GOAL";
    private static final String ENEMIES_GOAL = "ENEMIES_GOAL";
    private static final String TIME_GOAL = "TIME_GOAL";
    private static final int POINTS_THRESHOLD = 20000;
    private static final int ENEMIES_THRESHOLD = 3;
    private static final long TIME_THRESHOLD = 60000;

    private final Path store;
    private final Properties props = new Properties();
    
    private int runPoints = 0;
    private int runEnemies = 0;
    private long runTime = 0;

    public ApoMarioStateAchievements(java.nio.file.Path store) {
        this.store = store;
        load();
    }

    private void load() {
        try (FileInputStream in = new FileInputStream(store.toFile())) {
            props.load(in);
        } catch (IOException e) {
            // File might not exist
        }
    }

    public void beginRun() {
        runPoints = 0;
        runEnemies = 0;
        runTime = 0;
    }

    public void addPoints(int points) {
        runPoints += points;
        if (runPoints >= POINTS_THRESHOLD) {
            props.setProperty(POINTS_GOAL, "true");
        }
    }

    public void addEnemyKilled() {
        runEnemies++;
        if (runEnemies >= ENEMIES_THRESHOLD) {
            props.setProperty(ENEMIES_GOAL, "true");
        }
    }

    public void addTimeSurvived(long ms) {
        runTime = ms;
        if (runTime >= TIME_THRESHOLD) {
            props.setProperty(TIME_GOAL, "true");
        }
    }

    public void recordRunEnd(ApoMarioLevel level) {
        // Final evaluation check
        if (runPoints >= POINTS_THRESHOLD) props.setProperty(POINTS_GOAL, "true");
        if (runEnemies >= ENEMIES_THRESHOLD) props.setProperty(ENEMIES_GOAL, "true");
        if (runTime >= TIME_THRESHOLD) props.setProperty(TIME_GOAL, "true");
        saveAcrossSessions();
    }

    public List<String> getUnlockedAchievements() {
        List<String> unlocked = new ArrayList<>();
        if ("true".equals(props.getProperty(POINTS_GOAL))) unlocked.add(POINTS_GOAL);
        if ("true".equals(props.getProperty(ENEMIES_GOAL))) unlocked.add(ENEMIES_GOAL);
        if ("true".equals(props.getProperty(TIME_GOAL))) unlocked.add(TIME_GOAL);
        return unlocked;
    }

    public void saveAcrossSessions() {
        try (FileOutputStream out = new FileOutputStream(store.toFile())) {
            props.store(out, "ApoMario Achievements");
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}
```

package apoMario.game.panels;

import java.awt.Color;
import java.awt.Graphics2D;
import java.util.List;

import apoMario.game.ApoMarioPanel;

public class ApoMarioAchievementsPanel extends ApoMarioModelMenu {

    public ApoMarioAchievementsPanel(ApoMarioPanel game) {
        super(game);
    }

    @Override
    public void render(Graphics2D g) {
        super.renderBackgroundAndAnimation(g);
        
        g.setColor(Color.BLACK);
        g.drawString("Achievements", 100, 100);
        
        List<String> unlocked = getGame().getAchievements().getUnlockedAchievements();
        int y = 150;
        
        String[] goals = {"POINTS_GOAL", "ENEMIES_GOAL", "TIME_GOAL"};
        for (String goal : goals) {
            boolean isUnlocked = unlocked.contains(goal);
            g.drawString(goal + ": " + (isUnlocked ? "UNLOCKED" : "LOCKED"), 100, y);
            y += 30;
        }
    }

    @Override
    public void keyButtonReleased(int button, char character) {
        if (button == java.awt.event.KeyEvent.VK_ESCAPE) {
            getGame().setMenu();
        }
    }
    
    // Stub implementations for required interface/base class methods
    @Override public void mouseButtonFunction(String function) {}
    @Override public void mouseButtonReleased(int x, int y) {}
    @Override public boolean mouseDragged(int x, int y) { return false; }
    @Override public boolean mouseMoved(int x, int y) { return false; }
    @Override public boolean mousePressed(int x, int y, boolean bRight) { return false; }
    @Override public void think(int delta) {}
}
