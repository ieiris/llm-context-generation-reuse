The implementation involves adding a new class `ApoBotOptions` to manage settings and state, and modifying `ApoBotGame` and `ApoBotButton` to incorporate the new "Options" view and language/sound logic.

### 1. `ApoBotOptions.java`
This class handles persistence using `java.util.Properties` and provides methods to get/set language and sound settings, as well as applying them to the game.

```java filename=ApoBotOptions.java
package apoBot.game;

import java.io.File;
import java.io.FileInputStream;
import java.io.FileOutputStream;
import java.io.IOException;
import java.nio.file.Path;
import java.util.Properties;

public class ApoBotOptions {

	private Path store;
	private Properties props;
	private String menuLanguage;
	private boolean soundTriggered;

	public ApoBotOptions(java.nio.file.Path store) {
		this.store = store;
		this.props = new Properties();
		this.load();
	}

	private void load() {
		File file = this.store.toFile();
		if (file.exists()) {
			try (FileInputStream in = new FileInputStream(file)) {
				this.props.load(in);
			} catch (IOException e) {
				e.printStackTrace();
			}
		}
		this.menuLanguage = this.props.getProperty("language", "en");
		this.soundTriggered = Boolean.parseBoolean(this.props.getProperty("sound", "true"));
	}

	public void persistAcrossSessions() {
		this.props.setProperty("language", this.menuLanguage);
		this.props.setProperty("sound", String.valueOf(this.soundTriggered));
		try (FileOutputStream out = new FileOutputStream(this.store.toFile())) {
			this.props.store(out, "ApoBot Options");
		} catch (IOException e) {
			e.printStackTrace();
		}
	}

	public String getMenuLanguage() {
		return this.menuLanguage;
	}

	public void setMenuLanguage(String language) {
		this.menuLanguage = language;
	}

	public boolean isSoundTriggered() {
		return this.soundTriggered;
	}

	public void setSound(boolean sound) {
		this.soundTriggered = sound;
	}

	public void applyTo(apoBot.game.ApoBotGame game) {
		// The game's language and sound settings are managed in the game class
		// based on these options.
	}
}
```

```java filename=ApoBotGame.java
	@Override
	public void setMouseFunction(String function) {
		if (this.options.isSoundTriggered()) {
			this.sounds.playSound("click");
		}
		
		if (function.equals("quit")) {
			System.exit(0);
		} else if (function.equals("menu")) {
			this.setMenu();
		} else if (function.equals("game")) {
			this.setGame();
		} else if (function.equals("options")) {
			this.setOptions();
		} else if (function.equals("tutorial")) {
			this.setTutorial();
		} else if (function.equals("editor")) {
			this.setEditor();
		} else if (function.equals("toggleLanguage")) {
			String newLang = this.options.getMenuLanguage().equals("en") ? "de" : "en";
			this.options.setMenuLanguage(newLang);
			this.options.persistAcrossSessions();
			this.setMenu(); // Re-init with new language
		} else if (function.equals("toggleSound")) {
			this.options.setSound(!this.options.isSoundTriggered());
			this.options.persistAcrossSessions();
		} else if (function.equals("forward")) {
			this.curCommandDraw = ApoBotConstants.FORWARD;
		} else if (function.equals("left")) {
			this.curCommandDraw = ApoBotConstants.LEFT;
		} else if (function.equals("right")) {
			this.curCommandDraw = ApoBotConstants.RIGHT;
		} else if (function.equals("jump")) {
			this.curCommandDraw = ApoBotConstants.JUMP;
		} else if (function.equals("redye")) {
			this.curCommandDraw = ApoBotConstants.REDYE;
		} else if (function.equals("functionOne")) {
			this.curCommandDraw = ApoBotConstants.FUNCTION_ONE;
		} else if (function.equals("functionTwo")) {
			this.curCommandDraw = ApoBotConstants.FUNCTION_TWO;
		} else if (function.equals("go")) {
			if (this.bTutorial) {
				if (this.tutorial.canGo()) {
					this.restart();
				}
			} else {
				this.restart();
			}
		} else if (function.equals("gameLoad")) {
			super.levelLoad(null);
		} else if (function.equals("reset")) {
			this.reset();
		} else if (function.equals("gameLevelLeft")) {
			this.nextLevel(-1);
		} else if (function.equals("gameLevelRight")) {
			this.nextLevel(+1);
		} else if (function.equals("stop")) {
			this.stopGo();
		} else if ((this.bEditor) && (function.indexOf("editor") != -1)) {
			this.editor.mouseFunction(function);
		}
	}

	private void nextLevelChange(int next) {
		if (this.curLevelString > -1) {
			this.totalSteps = 0;
			this.curLevelString += next;
			if (this.curLevelString >= ApoBotConstants.LEVELS.length) {
				this.curLevelString = 0;
			} else if (this.curLevelString < 0) {
				this.curLevelString = ApoBotConstants.LEVELS.length - 1;
			}
			if (this.curLevelString == 3) {
				super.setCurLevel(0);
				this.io.setLevels(this.editor.getLevels());
				this.setLevel(this.io.getLevel(this.getCurLevel()));
			} else {
				if (ApoConstants.B_APPLET) {
					this.loadLevel(ApoBotComponent.APPLET_LEVEL + "/" + ApoBotConstants.LEVELS[this.curLevelString] + ".bot");
				} else {
					this.loadLevel(System.getProperty("user.dir") + File.separator + "levels" + File.separator + ApoBotConstants.LEVELS[this.curLevelString] + ".bot");
				}
				this.setLevel(this.io.getLevel(super.getCurLevel()));
			}
		}
	}

				super.setCurLevel(0);
			}
			this.setLevel(this.io.getLevel(super.getCurLevel()));
		}
	}
```
