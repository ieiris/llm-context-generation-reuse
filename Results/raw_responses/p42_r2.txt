package apoBot.game;

import java.io.*;
import java.util.Properties;
import java.nio.file.Path;

/**
 * Options mechanism to configure language and sound settings.
 * Persists settings to a file.
 */
public class ApoBotOptions {
    private Properties props = new Properties();
    private Path store;

    public ApoBotOptions(java.nio.file.Path store) {
        this.store = store;
        try (InputStream in = new FileInputStream(store.toFile())) {
            props.load(in);
        } catch (IOException e) {
            // Default values
            props.setProperty("language", "en");
            props.setProperty("sound", "true");
        }
    }

    public String getMenuLanguage() {
        return props.getProperty("language", "en");
    }

    public void setMenuLanguage(String language) {
        props.setProperty("language", language);
    }

    public boolean isSoundTriggered() {
        return Boolean.parseBoolean(props.getProperty("sound", "true"));
    }

    public void setSound(boolean sound) {
        props.setProperty("sound", String.valueOf(sound));
    }

    public void persistAcrossSessions() {
        try (OutputStream out = new FileOutputStream(store.toFile())) {
            props.store(out, null);
        } catch (IOException e) {
            e.printStackTrace();
        }
    }

    public void applyTo(apoBot.game.ApoBotGame game) {
        // Triggers UI update to reflect language change
        game.updateUI();
    }
}
