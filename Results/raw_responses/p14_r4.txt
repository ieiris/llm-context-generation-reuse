To integrate the Highscore feature into ApoMario, we will create a new `ApoMarioHighscore` class for logic and persistence, and an `ApoMarioHighscorePanel` for the UI. We will modify `ApoMarioLevel` to trigger the recording of game results, `ApoMarioPanel` to manage the highscore data and panel, and `ApoMarioMenu` to provide access to the highscore board.

### 1. ApoMarioHighscore.java
This class handles data management, file I/O, and sorting.

```java filename=ApoMarioHighscore.java
package apoMario.game.panels;

import java.io.*;
import java.nio.file.Path;
import java.util.*;
import apoMario.level.ApoMarioLevel;
import apoMario.entity.ApoMarioPlayer;

public class ApoMarioHighscore {

    private final Path store;
    private final List<ScoreEntry> scores = new ArrayList<>();

    public ApoMarioHighscore(Path store) {
        this.store = store;
        load();
    }

    private static class ScoreEntry {
        String name;
        int score;
        int time;

        ScoreEntry(String name, int score, int time) {
            this.name = name;
            this.score = score;
            this.time = time;
        }
    }

    public boolean storeRun(int score, int survivalTime, String playerName) {
        scores.add(new ScoreEntry(playerName, score, survivalTime));
        scores.sort((a, b) -> Integer.compare(b.score, a.score));
        persistAcrossRuns();
        return true;
    }

    public List<String> getPlayersNames() {
        List<String> names = new ArrayList<>();
        for (ScoreEntry entry : scores) names.add(entry.name);
        return names;
    }

    public List<Integer> getPlayersScores() {
        List<Integer> list = new ArrayList<>();
        for (ScoreEntry entry : scores) list.add(entry.score);
        return list;
    }

    public List<Integer> getSurvivalTimes() {
        List<Integer> list = new ArrayList<>();
        for (ScoreEntry entry : scores) list.add(entry.time);
        return list;
    }

    public void persistAcrossRuns() {
        try (PrintWriter writer = new PrintWriter(new BufferedWriter(new FileWriter(store.toFile())))) {
            for (ScoreEntry entry : scores) {
                writer.println(entry.name + "," + entry.score + "," + entry.time);
            }
        } catch (IOException e) {
            e.printStackTrace();
        }
    }

    private void load() {
        if (!store.toFile().exists()) return;
        try (BufferedReader reader = new BufferedReader(new FileReader(store.toFile()))) {
            String line;
            while ((line = reader.readLine()) != null) {
                String[] parts = line.split(",");
                if (parts.length == 3) {
                    scores.add(new ScoreEntry(parts[0], Integer.parseInt(parts[1]), Integer.parseInt(parts[2])));
                }
            }
            scores.sort((a, b) -> Integer.compare(b.score, a.score));
        } catch (IOException | NumberFormatException e) {
            e.printStackTrace();
        }
    }

    public void recordRunEnd(ApoMarioLevel level) {
        for (ApoMarioPlayer player : level.getPlayers()) {
            if (player.isBVisible()) {
                storeRun(player.getPoints(), level.getPassedTime() / 1000, player.getTeamName() != null ? player.getTeamName() : "Player");
            }
        }
    }
}
```

### 2. ApoMarioHighscorePanel.java
This class renders the highscore UI.

```java filename=ApoMarioHighscorePanel.java
package apoMario.game.panels;

import java.awt.*;
import java.awt.event.KeyEvent;
import java.awt.image.BufferedImage;
import java.util.ArrayList;
import apoMario.ApoMarioConstants;
import apoMario.game.ApoMarioPanel;

public class ApoMarioHighscorePanel extends ApoMarioModelMenu {

    public ApoMarioHighscorePanel(ApoMarioPanel game) {
        super(game);
    }

    @Override
    public void init() {
        super.init();
    }

    @Override
    public void makeBackground() {
        setIBackground(new BufferedImage(ApoMarioConstants.GAME_WIDTH, ApoMarioConstants.GAME_HEIGHT, BufferedImage.TYPE_INT_RGB));
        Graphics2D g = getIBackground().createGraphics();
        g.setColor(Color.DARK_GRAY);
        g.fillRect(0, 0, ApoMarioConstants.GAME_WIDTH, ApoMarioConstants.GAME_HEIGHT);
        g.dispose();
    }

    @Override
    public void makeBackgroundAnimation() {
        setBackgroundAnimation(new ArrayList<>());
    }

    @Override
    public void makeRunner() {
        setRunner(null);
    }

    @Override
    public void makeSearch() {
        setSearch(null);
    }

    @Override
    public void excecuteFunction() {
        if ("back".equals(getExcecuteFunction())) {
            getGame().setMenu();
        }
    }

    @Override
    public void keyButtonReleased(int button, char character) {
        if (button == KeyEvent.VK_ESCAPE) {
            getGame().setMenu();
        }
    }

    @Override
    public void mouseButtonFunction(String function) {
        setExcecuteFunction(function);
        excecuteFunction();
    }

    @Override
    public void mouseButtonReleased(int x, int y) {}

    @Override
    public boolean mouseMoved(int x, int y) { return false; }

    @Override
    public boolean mouseDragged(int x, int y) { return false; }

    @Override
    public boolean mousePressed(int x, int y, boolean bRight) { return false; }

    @Override
    public void think(int delta) {}

    @Override
    public void render(Graphics2D g) {
        renderBackgroundAndAnimation(g);
        g.setFont(ApoMarioConstants.FONT_MENU);
        g.setColor(Color.WHITE);
        g.drawString("Highscores", 50, 50);

        java.util.List<String> names = getGame().getHighscore().getPlayersNames();
        java.util.List<Integer> scores = getGame().getHighscore().getPlayersScores();
        java.util.List<Integer> times = getGame().getHighscore().getSurvivalTimes();

        g.setFont(ApoMarioConstants.FONT_STATISTICS);
        for (int i = 0; i < Math.min(10, names.size()); i++) {
            g.drawString((i + 1) + ". " + names.get(i) + " - " + scores.get(i) + " pts (" + times.get(i) + "s)", 50, 100 + i * 30);
        }
        
        g.drawString("Press ESC to go back", 50, ApoMarioConstants.GAME_HEIGHT - 50);
    }
}
```

### 3. ApoMarioLevel.java (Modified)
We update `setAnalysis` to trigger the highscore recording.

```java filename=ApoMarioLevel.java
package apoMario.level;

// ... existing imports ...
import apoMario.game.ApoMarioPanel;
// ... (keep all existing imports and fields)

public class ApoMarioLevel {
    // ... existing fields ...

    // ... existing methods ...

    public void setAnalysis(boolean bWin) {
        if (!this.bAnalysis) {
            return;
        }
        // RECORD HIGHSCORE
        if (this.component instanceof ApoMarioPanel) {
            ((ApoMarioPanel) this.component).getHighscore().recordRunEnd(this);
        }
        
        int addTimeDifficultyWidthPoints = (int)(ApoMarioConstants.POINT_TIME_MULTIPLICATOR * (float)this.getTime() + ApoMarioConstants.POINT_DIFFICULTY_MULTIPLICATOR * (float)this.getDifficulty() + ApoMarioConstants.POINT_WIDTH_MULTIPLICATOR * (float)this.getWidth());
        this.bAnalysis = false;
        this.component.setBWin(bWin);
        ApoMarioPanel game = ((ApoMarioPanel)(this.component));
        ApoMarioPlayer playerOne = this.getPlayers().get(0);
        ApoMarioPlayer playerTwo = this.getPlayers().get(1);
        float percentOne = (float)playerOne.getX() / ((float)ApoMarioConstants.TILE_SIZE * (float)this.getWidth());
        float percentTwo = (float)playerTwo.getX() / ((float)ApoMarioConstants.TILE_SIZE * (float)this.getWidth());
        if (!this.isOnePlayer()) {
            if (bWin) {
                if ((playerOne.getX() > playerTwo.getX()) && (playerOne.isBVisible())) {
                    playerOne.setPoints(playerOne.getPoints() + this.getFinish().getPoints() + addTimeDifficultyWidthPoints);
                    playerTwo.setPoints(playerTwo.getPoints() + (int)(addTimeDifficultyWidthPoints * percentTwo));
                } else if ((playerOne.getX() < playerTwo.getX()) && (playerTwo.isBVisible())) {
                    playerTwo.setPoints(playerTwo.getPoints() + this.getFinish().getPoints() + addTimeDifficultyWidthPoints);
                    playerOne.setPoints(playerOne.getPoints() + (int)(addTimeDifficultyWidthPoints * percentOne));
                } else if (playerOne.getX() == playerTwo.getX()) {
                    if (playerOne.getPoints() > playerTwo.getPoints()) {
                        playerOne.setPoints(playerOne.getPoints() + this.getFinish().getPoints() + addTimeDifficultyWidthPoints);
                        playerTwo.setPoints(playerTwo.getPoints() + (int)(addTimeDifficultyWidthPoints * percentTwo));
                    } else if (playerOne.getPoints() < playerTwo.getPoints()) {
                        playerTwo.setPoints(playerTwo.getPoints() + this.getFinish().getPoints() + addTimeDifficultyWidthPoints);
                        playerOne.setPoints(playerOne.getPoints() + (int)(addTimeDifficultyWidthPoints * percentOne));
                    } else {
                        if (playerOne.getCoins() > playerTwo.getCoins()) {
                            playerOne.setPoints(playerOne.getPoints() + this.getFinish().getPoints() + addTimeDifficultyWidthPoints);
                            playerTwo.setPoints(playerTwo.getPoints() + (int)(addTimeDifficultyWidthPoints * percentTwo));
                        } else if (playerOne.getCoins() < playerTwo.getCoins()) {
                            playerTwo.setPoints(playerTwo.getPoints() + this.getFinish().getPoints() + addTimeDifficultyWidthPoints);
                            playerOne.setPoints(playerOne.getPoints() + (int)(addTimeDifficultyWidthPoints * percentOne));
                        }
                    }
                }
            } else {
                if ((!playerOne.isBVisible()) && (playerTwo.isBVisible())) {
                    playerTwo.setPoints(playerTwo.getPoints() + ApoMarioConstants.POINTS_WIN_AGAINST_ANOTHER);
                } else if ((playerOne.isBVisible()) && (!playerTwo.isBVisible())) {
                    playerOne.setPoints(playerOne.getPoints() + ApoMarioConstants.POINTS_WIN_AGAINST_ANOTHER);
                }
            }
        } else {
            if (bWin) {
                playerOne.setPoints(playerOne.getPoints() + this.getFinish().getPoints() + addTimeDifficultyWidthPoints);
            }
        }
        if (game.isBEditor()) {
            game.setEditor();
        } else {
            game.setAnalysis();
        }
    }
    
    // ... (rest of the file remains unchanged)
}
```
