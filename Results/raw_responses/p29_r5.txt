package apoMario.game.panels;

import java.io.*;
import java.nio.file.Path;
import java.util.*;
import apoMario.level.ApoMarioLevel;

public class ApoMarioStateAchievements {
    public static final String POINTS_GOAL = "POINTS_GOAL";
    public static final String ENEMIES_GOAL = "ENEMIES_GOAL";
    public static final String TIME_GOAL = "TIME_GOAL";

    private final Path store;
    private Set<String> unlockedAchievements = new HashSet<>();
    private int currentPoints = 0;
    private int currentEnemiesKilled = 0;
    private long currentTimeSurvived = 0;

    public ApoMarioStateAchievements(Path store) {
        this.store = store;
        load();
    }

    public void beginRun() {
        this.currentPoints = 0;
        this.currentEnemiesKilled = 0;
        this.currentTimeSurvived = 0;
    }

    public void addPoints(int points) {
        this.currentPoints += points;
        if (this.currentPoints >= 20000) {
            unlock(POINTS_GOAL);
        }
    }

    public void addEnemyKilled() {
        this.currentEnemiesKilled++;
        if (this.currentEnemiesKilled >= 3) {
            unlock(ENEMIES_GOAL);
        }
    }

    public void addTimeSurvived(long ms) {
        this.currentTimeSurvived = ms;
        if (this.currentTimeSurvived >= 60000) {
            unlock(TIME_GOAL);
        }
    }

    public void recordRunEnd(ApoMarioLevel level) {
        if (this.currentPoints >= 20000) unlock(POINTS_GOAL);
        if (this.currentEnemiesKilled >= 3) unlock(ENEMIES_GOAL);
        if (this.currentTimeSurvived >= 60000) unlock(TIME_GOAL);
        saveAcrossSessions();
    }

    public List<String> getUnlockedAchievements() {
        return new ArrayList<>(unlockedAchievements);
    }

    public void saveAcrossSessions() {
        try (ObjectOutputStream oos = new ObjectOutputStream(new FileOutputStream(store.toFile()))) {
            oos.writeObject(unlockedAchievements);
        } catch (IOException e) {
            e.printStackTrace();
        }
    }

    private void load() {
        File file = store.toFile();
        if (file.exists()) {
            try (ObjectInputStream ois = new ObjectInputStream(new FileInputStream(file))) {
                unlockedAchievements = (Set<String>) ois.readObject();
            } catch (Exception e) {
                unlockedAchievements = new HashSet<>();
            }
        }
    }

    private void unlock(String achievement) {
        if (!unlockedAchievements.contains(achievement)) {
            unlockedAchievements.add(achievement);
            saveAcrossSessions();
        }
    }
}

import java.awt.Color;

        super(game);
    }

    @Override
    public void init() {}

    @Override
    public void keyButtonReleased(int button, char character) {
        if (button == KeyEvent.VK_ESCAPE) {
            getGame().setMenu();
        }
    }

    @Override
    public void mouseButtonFunction(String function) {
        if ("back".equals(function)) {
            getGame().setMenu();
        }
    }

    @Override
    public void mouseButtonReleased(int x, int y) {}

    @Override
    public boolean mouseMoved(int x, int y) { return false; }

    @Override
    public boolean mouseDragged(int x, int y) { return false; }

    @Override
    public boolean mousePressed(int x, int y, boolean bRight) { return false; }

    @Override
    public void think(int delta) {}

    @Override
    public void render(Graphics2D g) {
        g.setColor(Color.WHITE);
        g.fillRect(0, 0, 800, 600);
        g.setColor(Color.BLACK);
        g.drawString("Achievements", 350, 50);

        List<String> unlocked = getGame().getAchievements().getUnlockedAchievements();
        String[] all = {ApoMarioStateAchievements.POINTS_GOAL, ApoMarioStateAchievements.ENEMIES_GOAL, ApoMarioStateAchievements.TIME_GOAL};

        for (int i = 0; i < all.length; i++) {
            g.drawString(all[i] + ": " + (unlocked.contains(all[i]) ? "UNLOCKED" : "LOCKED"), 100, 100 + i * 50);
        }
        g.drawString("Press ESC to return", 100, 400);
    }
}
