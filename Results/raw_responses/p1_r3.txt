package apoMario.game.panels;

import java.awt.Color;
import java.awt.Graphics2D;
import java.io.FileInputStream;

import apoMario.ApoMarioConstants;
import apoMario.game.ApoMarioPanel;
import apoMario.level.ApoMarioLevel;
import org.apogames.help.ApoHelp;

public class ApoMarioHighscore extends ApoMarioModel {

    private final Path store;
    private List<String> names;
    private List<Integer> scores;
    private List<Integer> times;

    public ApoMarioHighscore(Path store) {
        this.store = store;
        this.names = new ArrayList<>();
        this.scores = new ArrayList<>();
        this.times = new ArrayList<>();
        load();
    }

    @Override
    public void init() {
    }

    @Override
    public void keyButtonReleased(int button, char character) {
        if (button == java.awt.event.KeyEvent.VK_ESCAPE) {
            this.getGame().setMenu();
        }
    }

    @Override
    public void mouseButtonFunction(String function) {
        if (function.equals("highscore_back")) {
            this.getGame().setMenu();
        }
    }

    @Override
    public void mouseButtonReleased(int x, int y) {
    }

    @Override
    public boolean mouseMoved(int x, int y) {
        return false;
    }

    @Override
    public boolean mouseDragged(int x, int y) {
        return false;
    }

    @Override
    public boolean mousePressed(int x, int y, boolean bRight) {
        return false;
    }

    @Override
    public void think(int delta) {
    }

    @Override
    public void render(Graphics2D g) {
        g.setColor(Color.WHITE);
        g.fillRect(0, 0, ApoMarioConstants.GAME_WIDTH, ApoMarioConstants.GAME_HEIGHT);
        g.setColor(Color.BLACK);
        g.setFont(ApoMarioConstants.FONT_MENU);
        
        g.drawString("Highscore", 50, 50);
        g.setFont(ApoMarioConstants.FONT_STATISTICS);
        
        for (int i = 0; i < Math.min(10, names.size()); i++) {
            String entry = (i + 1) + ". " + names.get(i) + " - " + scores.get(i) + " pts - " + ApoHelp.getTimeToDraw(times.get(i));
            g.drawString(entry, 50, 100 + i * 30);
        }
    }

    public boolean storeRun(int score, int survivalTime, String playerName) {
        names.add(playerName != null ? playerName : "Unknown");
        scores.add(score);
        times.add(survivalTime);
        sortHighscores();
        return true;
    }

    public List<String> getPlayersNames() {
        return names;
    }

    public List<Integer> getPlayersScores() {
        return scores;
    }

    public List<Integer> getSurvivalTimes() {
        return times;
    }

    public void persistAcrossRuns() {
        try (ObjectOutputStream oos = new ObjectOutputStream(new FileOutputStream(store.toFile()))) {
            oos.writeObject(names);
            oos.writeObject(scores);
            oos.writeObject(times);
        } catch (IOException e) {
            e.printStackTrace();
        }
    }

    public void recordRunEnd(ApoMarioLevel level) {
        if (level.getPlayers() != null && !level.getPlayers().isEmpty()) {
            storeRun(level.getPlayers().get(0).getPoints(), level.getPassedTime(), level.getPlayers().get(0).getTeamName());
            persistAcrossRuns();
        }
    }

    @SuppressWarnings("unchecked")
    private void load() {
        if (store.toFile().exists()) {
            try (ObjectInputStream ois = new ObjectInputStream(new FileInputStream(store.toFile()))) {
                names = (List<String>) ois.readObject();
                scores = (List<Integer>) ois.readObject();
                times = (List<Integer>) ois.readObject();
            } catch (Exception e) {
                e.printStackTrace();
            }
        }
    }

    private void sortHighscores() {
        List<Integer> indices = new ArrayList<>();
        for (int i = 0; i < scores.size(); i++) indices.add(i);
        Collections.sort(indices, new Comparator<Integer>() {
            @Override
            public int compare(Integer i1, Integer i2) {
                return scores.get(i2).compareTo(scores.get(i1));
            }
        });

        List<String> sortedNames = new ArrayList<>();
        List<Integer> sortedScores = new ArrayList<>();
        List<Integer> sortedTimes = new ArrayList<>();
        for (int index : indices) {
            sortedNames.add(names.get(index));
            sortedScores.add(scores.get(index));
            sortedTimes.add(times.get(index));
        }
        names = sortedNames;
        scores = sortedScores;
        times = sortedTimes;
    }
}
