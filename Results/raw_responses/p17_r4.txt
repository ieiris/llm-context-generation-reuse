    public static final String BUTTON_BACK = "back";
    public static final String POINTS_GOAL = "POINTS_GOAL";
    public static final String ENEMIES_GOAL = "ENEMIES_GOAL";
    public static final String TIME_GOAL = "TIME_GOAL";

    private final Path store;
    private final List<String> unlockedAchievements;
    private int currentRunPoints;
    private int currentRunEnemies;
    private long currentRunTime;

    public ApoMarioStateAchievements(Path store) {
        super(null);
        this.store = store;
        this.unlockedAchievements = new ArrayList<>();
        this.loadUnlockedAchievements();
    }

    public void setGame(ApoMarioPanel game) {
        this.game = game;
    }

    private void loadUnlockedAchievements() {
        if (Files.exists(store)) {
            try (BufferedReader reader = Files.newBufferedReader(store)) {
                String line;
                while ((line = reader.readLine()) != null) {
                    unlockedAchievements.add(line);
                }
            } catch (IOException e) {
                e.printStackTrace();
            }
        }
    }

    public void saveAcrossSessions() {
        try (BufferedWriter writer = Files.newBufferedWriter(store)) {
            for (String achievement : unlockedAchievements) {
                writer.write(achievement);
                writer.newLine();
            }
        } catch (IOException e) {
            e.printStackTrace();
        }
    }

    public void beginRun() {
        this.currentRunPoints = 0;
        this.currentRunEnemies = 0;
        this.currentRunTime = 0;
    }

    public void addPoints(int points) {
        this.currentRunPoints += points;
        if (this.currentRunPoints >= 20000 && !unlockedAchievements.contains(POINTS_GOAL)) {
            unlock(POINTS_GOAL);
        }
    }

    public void addEnemyKilled() {
        this.currentRunEnemies++;
        if (this.currentRunEnemies >= 3 && !unlockedAchievements.contains(ENEMIES_GOAL)) {
            unlock(ENEMIES_GOAL);
        }
    }

    public void addTimeSurvived(long ms) {
        this.currentRunTime += ms;
        if (this.currentRunTime >= 60000 && !unlockedAchievements.contains(TIME_GOAL)) {
            unlock(TIME_GOAL);
        }
    }

    public void recordRunEnd(ApoMarioLevel level) {
        int maxPoints = 0;
        for (ApoMarioPlayer player : level.getPlayers()) {
            if (player.getPoints() > maxPoints) {
                maxPoints = player.getPoints();
            }
        }
        if (maxPoints >= 20000 && !unlockedAchievements.contains(POINTS_GOAL)) {
            unlock(POINTS_GOAL);
        }
        if (currentRunEnemies >= 3 && !unlockedAchievements.contains(ENEMIES_GOAL)) {
            unlock(ENEMIES_GOAL);
        }
        if (currentRunTime >= 60000 && !unlockedAchievements.contains(TIME_GOAL)) {
            unlock(TIME_GOAL);
        }
    }

    private void unlock(String id) {
        if (!unlockedAchievements.contains(id)) {
            unlockedAchievements.add(id);
            saveAcrossSessions();
        }
    }

    public List<String> getUnlockedAchievements() {
        return new ArrayList<>(unlockedAchievements);
    }

    @Override
    public void init() {
        this.makeBackground();
        this.makeRunner();
        this.makeBackgroundAnimation();
        this.makeSearch();
    }

    @Override

        g.dispose();
    }

        this.setBackgroundAnimation(new ArrayList<>());
    }

        this.setRunner(null);
    }

    @Override
    public void makeSearch() {
        this.setSearch(null);
    }

    @Override
    public void excecuteFunction() {
        if (this.getExcecuteFunction().equals(BUTTON_BACK)) {
            this.getGame().setMenu();
        }
    }

    @Override
    public void keyButtonReleased(int button, char character) {
        if (button == KeyEvent.VK_ESCAPE) {
            this.getGame().setMenu();
        }
    }

    @Override
    public void mouseButtonFunction(String function) {
        if (function.equals(BUTTON_BACK)) {
            this.getGame().setMenu();
        }
    }

    @Override
    public void mouseButtonReleased(int x, int y) {}

    @Override
    public boolean mouseMoved(int x, int y) { return false; }

    @Override
    public boolean mouseDragged(int x, int y) { return false; }

    @Override
    public boolean mousePressed(int x, int y, boolean bRight) { return false; }

    public void think(int delta) {}

        g.drawImage(this.getIBackground(), 0, 0, null);
        g.setColor(Color.BLACK);
        g.setFont(ApoMarioConstants.FONT_STATISTICS);

            String status = unlockedAchievements.contains(achievement) ? "[UNLOCKED] " : "[LOCKED] ";
            g.drawString(status + achievement, 50, y);

    }

        if (!((this.getX() + 3 * ApoMarioConstants.TILE_SIZE > p.x) && (this.getX() - 3 * ApoMarioConstants.TILE_SIZE < p.y))) { this.setBVisible(false); return; }

        super.think(delta, level);
        this.thinkSpeedAndMoveEnemy(delta, level);
        this.thinkCollision(delta, level);
    }
