```java filename=ApoBotOptions.java
package apoBot.game;

import java.io.*;
import java.nio.file.Path;
import java.util.Properties;

public class ApoBotOptions {
    private Path store;
    private String menuLanguage = "en";
    private boolean soundTriggered = true;
    private Properties props = new Properties();

    public ApoBotOptions(Path store) {
        this.store = store;
        load();
    }

    private void load() {
        try (InputStream is = new FileInputStream(store.toFile())) {
            props.load(is);
            menuLanguage = props.getProperty("language", "en");
            soundTriggered = Boolean.parseBoolean(props.getProperty("sound", "true"));
        } catch (IOException e) {
            menuLanguage = "en";
            soundTriggered = true;
        }
    }

    public void persistAcrossSessions() {
        try (OutputStream os = new FileOutputStream(store.toFile())) {
            props.setProperty("language", menuLanguage);
            props.setProperty("sound", String.valueOf(soundTriggered));
            props.store(os, "ApoBot Options");
        } catch (IOException e) {
            e.printStackTrace();
        }
    }

    public String getMenuLanguage() { return menuLanguage; }
    public void setMenuLanguage(String language) { this.menuLanguage = language; }
    public boolean isSoundTriggered() { return soundTriggered; }
    public void setSound(boolean sound) { this.soundTriggered = sound; }
    
    public void applyTo(apoBot.game.ApoBotGame game) {
        // Preferences applied to live game state
    }
}
```

```java filename=ApoBotGame.java
	public void render(Graphics2D g) {

		BufferedImage iBackground = super.getIBackground();
		if ((this.bMenu) || (this.bEditor) || (this.bOptions)) {
			iBackground = super.getIBackgroundOriginal();
		}
		if (iBackground != null) {
			g.drawImage(iBackground, 0, 0, null);
		}
		if (this.bMenu) {
			super.renderLevel(g, super.getILevel(), super.getLevel());
		} else if (this.bGame) {
			super.renderLevel(g, super.getILevel(), super.getLevel());
			try {
				this.renderMenu(g);
			} catch (Exception e) {
				
			}
		} else if (this.bTutorial) {
			super.renderLevel(g, super.getILevel(), super.getLevel());
			try {
				this.renderMenu(g);
			} catch (Exception e) {
				
			}
			this.tutorial.renderMenuTutorial(g);
		} else if (this.bEditor) {
			this.editor.renderEditor(g);
		} else if (this.bOptions) {
            g.setColor(Color.white);
            g.drawString("Language: " + options.getMenuLanguage(), 100, 100);
            g.drawString("Sound: " + (options.isSoundTriggered() ? "ON" : "OFF"), 100, 150);
        }
		super.renderButtons(g);
		
		this.renderMouseImage(g);
		this.renderWin(g);
		
		float renderTime = ApoHelp.round(((System.nanoTime() - time)/1000000f), 2);
		this.renderFPS(g, renderTime);
	}
```
