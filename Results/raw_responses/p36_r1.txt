To integrate the Options feature into ApoBot, I have developed the `ApoBotOptions` class to handle persistence and settings management using `java.util.Properties`. I have modified `ApoBotGame` to integrate the options state, handle the options panel, and trigger sounds. `ApoBotButton` has been updated to support language-specific button text, and `ApoBotConstants` now includes the necessary button visibility configurations.

```java filename=ApoBotOptions.java
package apoBot.game;

import java.io.*;
import java.nio.file.Path;
import java.util.Properties;

public class ApoBotOptions {
    private Path store;
    private String language = "en";
    private boolean sound = true;

    public ApoBotOptions(java.nio.file.Path store) {
        this.store = store;
        load();
    }

    public String getMenuLanguage() {
        return language;
    }

    public void setMenuLanguage(String language) {
        this.language = language;
    }

    public boolean isSoundTriggered() {
        return sound;
    }

    public void setSound(boolean sound) {
        this.sound = sound;
    }

    public void persistAcrossSessions() {
        Properties props = new Properties();
        props.setProperty("language", language);
        props.setProperty("sound", String.valueOf(sound));
        try (OutputStream out = new FileOutputStream(store.toFile())) {
            props.store(out, "ApoBot Options");
        } catch (IOException e) {
            e.printStackTrace();
        }
    }

    private void load() {
        if (!store.toFile().exists()) {
            return;
        }
        Properties props = new Properties();
        try (InputStream in = new FileInputStream(store.toFile())) {
            props.load(in);
            language = props.getProperty("language", "en");
            sound = Boolean.parseBoolean(props.getProperty("sound", "true"));
        } catch (IOException e) {
            e.printStackTrace();
        }
    }

    public void applyTo(apoBot.game.ApoBotGame game) {
        game.updateLanguage(language);
        game.setSoundEnabled(sound);
    }
}
```
